	print('Введите коэффициенты квадратного уравннения a*(x^2)b*x+c=0: ')
	a = float(input('a = '))
	b = float(input('b = '))
	c = float(input('c = '))

	d = b**2 - 4*a*c

	if d>0:
    		print('x1 = ', (-b + d**0.5) / (2*a))
    		print('x2 = ', (-b - d**0.5) / (2*a))

    		print('x = ', (-b) / (2*a))
